4. ```
ENQUEUE(Q,x)
    tail = Q.tail
    if tail == Q.length
        tail = 1
    else
        tail = tail + 1
    if tail != Q.head
         Q[Q.tail] = x
         Q.tail=tail

DEQUEUE(Q)
    if Q.head != Q.tail
        x = Q[Q.head]
        if Q.head == Q.length
            Q.head = 1
        else
            Q.head = Q.head + 1
        return x
