    declare r[n]
    declare stack s //栈s里保存的就是切割方案
    for i=0 to n
       r[i]=-INMAX
    return MEMOIZED-CUT-ROD-AUX(p,n,r,s)

MEMOIZED-CUT-ROD-AUX(p,n,r,s)
   if r[n] >= 0
       return r[n]
   if n == 0
       q = 0
   else
        q = -INTMAX
        for i=1 to n
          tempq = p[i] +MEMOIZED-CUT-ROD-AUX(p,n-i,r)
          if tempq > q then
              q=tempq
              k=i
        s.push(k)
   r[n]=q
   return q
